\begin{eqnarray*}
  gcd(n, m)  = g &\mbox{ iff }& \mbox{$g$ divides both $n$ and $m$, and}\\
  && \mbox{for each $i > g$, $i$ does not divide both $n$ and $m$.}
\end{eqnarray*}
